<script>
const BossNames = Array.from({ length: 14 }, (_, i) => `Boss${i + 1}`);

let currentBossIndex = 0;
let currentBossUsername = "???";


// Function to update the displayed boss sprite and name
function showCurrentBoss() {
  const bossesDiv = document.getElementById('boss-bosses');
  bossesDiv.innerHTML = `
    <img src="assets/images/${BossNames[currentBossIndex]}.png" alt="Boss Sprite" class="boss-sprite" />
  `;
  document.getElementById("boss-name").innerHTML = `
    <span class="boss-username">${currentBossUsername}</span>
  `;
}

// Example: Call this function whenever the boss resets
showCurrentBoss();

// Example: Listen for boss reset from backend
socket.on("bossReset", ({ newBossIndex, newBossUsername }) => {
  currentBossIndex = newBossIndex;
  currentBossUsername = newBossUsername;
  showCurrentBoss();
});
</script>
